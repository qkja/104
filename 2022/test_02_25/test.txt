#define _CRT_SECURE_NO_WARNINGS 1

#include<stdio.h>
#include<string.h>
#include<errno.h>

//Абtest.c ЕФФкШнПНБДЕН test.txtжа
int main()
{

	FILE* wpf = fopen("test.txt", "w");

	FILE* rpf = fopen("test.c", "r");

	if (rpf == NULL)
	{
		printf("%s\n", strerror(errno));
		fclose(wpf);
		wpf = NULL;
		return 1;
	}

	//ПНБД
	int ch = 0;
	while ((ch = fgetc(rpf)) != EOF)
	{
		fputc(ch, wpf);
	}
	
    //ЙиБеЮФМў
	fclose(rpf);  
	rpf = NULL;
	fclose(wpf);
	wpf = NULL;
	return 0;
}
